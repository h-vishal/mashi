#Core Reference - mashi.application

Mashi Application

#####Syntax

	var INSTANCE = new mashi.application();

#####Configuration and default values

	INSTANCE.config(
	
	  meta : {
        namespace: 'MYAPP', // required - Name of the instance
        title: 'none', // optional
        author: 'unknown' // optional
      },
	  
	  style : {
  	    backgroundImage: 'none', // optional - CSS notation
  	    backgroundColor: '#FFFFFF' // optional - CSS notation
      },
	  
	  animation : {
        fade : {
          duration: 300, // optional (miliseconds)
	      steps: 20, // optional
	      start: 0, // optional
	      end: 1 // optional
        },
        canvas: false, // optional - false/true
        start: false, // optional - false/true
        rewind: false, // optional - false/true
        accessible: 'text' // optional - text/stage
      },

      /* for module mashi.controls */
	  buttons : { 
        type: false, // optional - false/single/movie/slideshow/all
        position : {
          vertical: 'bottom', // optional - bottom/middle/top
          horizontal: 'right' // optional - right/center/left
        }
      }
	);


#####Adding a frame

	INSTANCE.add(DURATION, CALLBACK);

#####Editing the frame

	INSTANCE.set(PARAM-OBJECT);

#####Movie controls

	INSTANCE.stop();
	
	INSTANCE.first();
	
	INSTANCE.prev();
	
	INSTANCE.next();
	
	INSTANCE.flipToNumber(FRAME-NUMBER);

#####Example
	var app = new mashi.application();
	
	timeline.add(2000, function() {
      app.set({
        html: 'Frame one',
        style: {
          backgroundColor: "red"
        }
      });
	});
	
	timeline.add(2000, function() {
      app.set({
        html: 'Frame two'
      });
	});

	window.onload = function() {
	  app.config({
	    meta: {
	      namespace: 'app',
	      title: 'Example App',
	      author: 'Alex Ahorn'
	    },
	    buttons: {
	      type: 'single'
	    },
	    animation: {
	      start: true
	    }
	  });
	};
